// Parametrized Content Addressable Memory (CAM) Module
module cam #(
    parameter DATA_WIDTH = 8,   // Width of each data word
    parameter DEPTH = 16        // Number of entries in the CAM
)(
    input  wire                     clk,
    input  wire                     rst_n,
    // Write interface
    input  wire                     wr_en,      // Write enable
    input  wire [$clog2(DEPTH)-1:0] wr_addr,    // Write address
    input  wire [DATA_WIDTH-1:0]    wr_data,    // Data to write
    // Search interface
    input  wire                     search_en,  // Search enable
    input  wire [DATA_WIDTH-1:0]    search_data,// Data to search
    output reg                      match,      // Match found flag
    output reg [$clog2(DEPTH)-1:0]  match_addr  // Address of matched data
);

    // Internal memory array
    reg [DATA_WIDTH-1:0] mem [0:DEPTH-1];

    integer i;

    // Write operation
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            for (i = 0; i < DEPTH; i = i + 1)
                mem[i] <= {DATA_WIDTH{1'b0}};
        end else if (wr_en) begin
            mem[wr_addr] <= wr_data;
        end
    end

    // Search operation
    always @(*) begin
        match = 1'b0;
        match_addr = {($clog2(DEPTH)){1'b0}};
        for (i = 0; i < DEPTH; i = i + 1) begin
            if (search_en && (mem[i] == search_data)) begin
                match = 1'b1;
                match_addr = i[$clog2(DEPTH)-1:0];
            end
        end
    end

endmodule
