module decoder_n #(parameter N = 8)(
    input  [$clog2(N)-1:0] a,   // Binary input
    output reg [N-1:0]     y    // One-hot output
);
    integer i;
    always @(*) begin
        y = 0;
        for (i = 0; i < N; i = i + 1) begin
            if (a == i[$clog2(N)-1:0])
                y[i] = 1;
        end
    end
endmodule
