module mux #(
    parameter integer N = 4,                // Number of inputs
    parameter integer DATA_WIDTH = 8        // Width of each input
)(
    input  wire [N*DATA_WIDTH-1:0] data_in, // Concatenated input buses
    input  wire [$clog2(N)-1:0]    sel,     // Select signal
    output wire [DATA_WIDTH-1:0]   data_out // Output
);

    // Internal: Unpack data_in into array for easier indexing
    wire [DATA_WIDTH-1:0] data_array [0:N-1];
    genvar i;
    generate
        for (i = 0; i < N; i = i + 1) begin : UNPACK_INPUTS
            assign data_array[i] = data_in[(i+1)*DATA_WIDTH-1 : i*DATA_WIDTH];
        end
    endgenerate

    // Multiplexor logic
    assign data_out = data_array[sel];

endmodule
