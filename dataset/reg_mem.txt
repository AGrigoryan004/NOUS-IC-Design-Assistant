module reg_mem #
(
    parameter ADDR_WIDTH = 4,      // Address bus width (memory depth = 2^ADDR_WIDTH)
    parameter DATA_WIDTH = 8       // Data bus width
)
(
    input  wire                   clk,        // System clock
    input  wire                   rst_n,      // Active-low reset
    input  wire                   wr_en,      // Write enable (active high)
    input  wire                   rd_en,      // Read enable (active high)
    input  wire [ADDR_WIDTH-1:0]  addr,       // Address bus
    input  wire [DATA_WIDTH-1:0]  wr_data,    // Data input (for write)
    output reg  [DATA_WIDTH-1:0]  rd_data,    // Data output (for read)
    output reg                    ready       // Ready signal: 1 when rd_data is valid
);

    // Internal register array
    reg [DATA_WIDTH-1:0] reg_array [0:(1<<ADDR_WIDTH)-1];

    integer i;

    // Synchronous operations and reset
    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            rd_data <= {DATA_WIDTH{1'b0}};
            ready   <= 1'b0;
            for (i = 0; i < (1<<ADDR_WIDTH); i = i + 1)
                reg_array[i] <= {DATA_WIDTH{1'b0}};
        end else begin
            ready <= 1'b0;
            if (wr_en) begin
                reg_array[addr] <= wr_data;
                ready <= 1'b0;
            end else if (rd_en) begin
                rd_data <= reg_array[addr];
                ready   <= 1'b1;
            end else begin
                ready <= 1'b0;
            end
        end
    end

endmodule
