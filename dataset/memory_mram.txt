module mram_memory #(
    parameter ADDR_WIDTH = 8,      // Address width (number of address bits)
    parameter DATA_WIDTH = 8       // Data width (bits per memory word)
)(
    input wire clk,                // Clock
    input wire rst_n,              // Active-low reset
    input wire [ADDR_WIDTH-1:0] addr, // Address bus
    input wire [DATA_WIDTH-1:0] din,  // Data input
    output reg [DATA_WIDTH-1:0] dout, // Data output
    input wire read,               // Read enable
    input wire write,              // Write enable
    output reg busy                // Busy flag (operation in progress)
);

    // Memory array
    reg [DATA_WIDTH-1:0] mem [0:(1<<ADDR_WIDTH)-1];

    // Internal state machine
    typedef enum reg [1:0] {
        IDLE = 2'b00,
        WRITE = 2'b01
    } state_t;

    state_t state;
    reg [0:0] op_cnt; // Operation counter for write

    integer i;

    // MRAM characteristics:
    // - Write: takes 1 cycle
    // - Read: combinational (no delay)
    // - No erase operation (non-volatile, direct overwrite)

    always @(posedge clk or negedge rst_n) begin
        if (!rst_n) begin
            state <= IDLE;
            busy <= 0;
            dout <= 0;
            op_cnt <= 0;
            // Optional: initialize memory to 0
            for (i = 0; i < (1<<ADDR_WIDTH); i = i + 1)
                mem[i] <= {DATA_WIDTH{1'b0}};
        end else begin
            case (state)
                IDLE: begin
                    busy <= 0;
                    if (write) begin
                        state <= WRITE;
                        busy <= 1;
                        op_cnt <= 1; // Write takes 1 cycle
                    end else if (read) begin
                        dout <= mem[addr];
                    end
                end
                WRITE: begin
                    if (op_cnt > 1) begin
                        op_cnt <= op_cnt - 1;
                    end else begin
                        mem[addr] <= din;
                        state <= IDLE;
                        busy <= 0;
                    end
                end
                default: state <= IDLE;
            endcase
        end
    end

endmodule
