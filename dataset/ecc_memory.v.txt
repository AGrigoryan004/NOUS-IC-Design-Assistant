module ecc_memory #(
    parameter DATA_WIDTH = 8,           // Data width (e.g., 8 or 16)
    parameter ADDR_WIDTH = 4            // Address width (memory depth = 2^ADDR_WIDTH)
)(
    input                      clk,
    input                      rst,
    input                      we,      // Write enable
    input                      re,      // Read enable
    input  [ADDR_WIDTH-1:0]    addr,
    input  [DATA_WIDTH-1:0]    din,     // Data input
    output reg [DATA_WIDTH-1:0] dout,   // Data output (corrected)
    output reg                 sec,     // Single error corrected
    output reg                 ded      // Double error detected
);

    // Calculate number of parity bits for SECDED (Hamming)
    function integer calc_parity_bits;
        input integer d;
        integer p;
        begin
            p = 1;
            while ((2**p) < (d + p + 1)) p = p + 1;
            calc_parity_bits = p;
        end
    endfunction

    localparam PARITY_BITS = calc_parity_bits(DATA_WIDTH);
    localparam ECC_WIDTH   = DATA_WIDTH + PARITY_BITS + 1; // +1 for overall parity

    // Memory array: stores data + ECC bits
    reg [ECC_WIDTH-1:0] mem [0:(1<<ADDR_WIDTH)-1];

    // ECC encode function (returns [DATA+PARITY+1:0])
    function [ECC_WIDTH-1:0] ecc_encode;
        input [DATA_WIDTH-1:0] data;
        integer i, j, k;
        reg [ECC_WIDTH-1:0] codeword;
        reg [PARITY_BITS-1:0] parity;
        reg overall_parity;
        begin
            // Place data bits in codeword (skip parity bit positions)
            codeword = 0;
            k = 0;
            for (i = 1; i <= DATA_WIDTH + PARITY_BITS; i = i + 1) begin
                if (^i === 1'b1) // If i is power of 2 (parity bit position)
                    codeword[i-1] = 0;
                else begin
                    codeword[i-1] = data[k];
                    k = k + 1;
                end
            end
            // Calculate parity bits
            for (i = 0; i < PARITY_BITS; i = i + 1) begin
                parity[i] = 0;
                for (j = 1; j <= DATA_WIDTH + PARITY_BITS; j = j + 1) begin
                    if (j & (1 << i))
                        parity[i] = parity[i] ^ codeword[j-1];
                end
                codeword[(1<<i)-1] = parity[i];
            end
            // Calculate overall parity (SECDED)
            overall_parity = ^codeword[DATA_WIDTH + PARITY_BITS - 1:0];
            codeword[DATA_WIDTH + PARITY_BITS] = overall_parity;
            ecc_encode = codeword;
        end
    endfunction

    // ECC decode and correct function
    function [DATA_WIDTH-1:0] ecc_decode;
        input [ECC_WIDTH-1:0] codeword;
        output reg sec, ded;
        integer i, j, k;
        reg [PARITY_BITS-1:0] syndrome;
        reg overall_parity, calc_overall_parity;
        reg [DATA_WIDTH-1:0] data_out;
        reg [ECC_WIDTH-2:0] codeword_no_overall;
        begin
            // Calculate syndrome
            for (i = 0; i < PARITY_BITS; i = i + 1) begin
                syndrome[i] = 0;
                for (j = 1; j <= DATA_WIDTH + PARITY_BITS; j = j + 1) begin
                    if (j & (1 << i))
                        syndrome[i] = syndrome[i] ^ codeword[j-1];
                end
            end
            // Calculate overall parity
            calc_overall_parity = ^codeword[DATA_WIDTH + PARITY_BITS - 1:0];
            overall_parity = codeword[DATA_WIDTH + PARITY_BITS];
            // Error detection
            if ((syndrome != 0) && (overall_parity != calc_overall_parity)) begin
                // Single-bit error: correctable
                sec = 1;
                ded = 0;
                codeword_no_overall = codeword[DATA_WIDTH + PARITY_BITS - 1:0];
                codeword_no_overall[{syndrome} - 1] = ~codeword_no_overall[{syndrome} - 1];
            end else if ((syndrome != 0) && (overall_parity == calc_overall_parity)) begin
                // Double-bit error: detected, not correctable
                sec = 0;
                ded = 1;
                codeword_no_overall = codeword[DATA_WIDTH + PARITY_BITS - 1:0];
            end else begin
                // No error
                sec = 0;
                ded = 0;
                codeword_no_overall = codeword[DATA_WIDTH + PARITY_BITS - 1:0];
            end
            // Extract data bits
            k = 0;
            for (i = 1; i <= DATA_WIDTH + PARITY_BITS; i = i + 1) begin
                if (!(^i === 1'b1)) begin
                    data_out[k] = codeword_no_overall[i-1];
                    k = k + 1;
                end
            end
            ecc_decode = data_out;
        end
    endfunction

    // Internal signals for read
    reg [ECC_WIDTH-1:0] read_codeword;
    reg [DATA_WIDTH-1:0] corrected_data;
    reg sec_int, ded_int;

    // Write operation
    always @(posedge clk) begin
        if (rst) begin
            // Optional: clear memory
        end else if (we) begin
            mem[addr] <= ecc_encode(din);
        end
    end

    // Read operation
    always @(posedge clk) begin
        if (rst) begin
            dout <= 0;
            sec  <= 0;
            ded  <= 0;
        end else if (re) begin
            read_codeword = mem[addr];
            corrected_data = ecc_decode(read_codeword, sec_int, ded_int);
            dout <= corrected_data;
            sec  <= sec_int;
            ded  <= ded_int;
        end
    end

    // For testbench: expose memory for error injection
    // synthesis translate_off
    function [ECC_WIDTH-1:0] get_codeword;
        input [ADDR_WIDTH-1:0] addr_in;
        begin
            get_codeword = mem[addr_in];
        end
    endfunction
    task set_codeword;
        input [ADDR_WIDTH-1:0] addr_in;
        input [ECC_WIDTH-1:0] codeword_in;
        begin
            mem[addr_in] = codeword_in;
        end
    endtask
    // synthesis translate_on

endmodule
