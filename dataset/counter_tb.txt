module counter_tb;

    // Parameters for test
    parameter WIDTH = 4;
    parameter MAX_VALUE = 9;

    // Testbench signals
    reg clk;
    reg rst_n;
    reg en;
    wire [WIDTH-1:0] count;
    wire wrap;

    integer i;

    // Instantiate the counter
    counter #(
        .WIDTH(WIDTH),
        .MAX_VALUE(MAX_VALUE)
    ) uut (
        .clk(clk),
        .rst_n(rst_n),
        .en(en),
        .count(count),
        .wrap(wrap)
    );

    // Clock generation: 10ns period
    initial clk = 0;
    always #5 clk = ~clk;

    // Task to check counter value
    task check_count;
        input [WIDTH-1:0] expected;
        input string msg;
        begin
            if (count !== expected) begin
                $display("ERROR: %s At time %0t, count=%0d, expected=%0d", msg, $time, count, expected);
            end
        end
    endtask

    initial begin
        $display("Starting parameterized counter testbench at %0t", $time);

        // Initialize signals
        rst_n = 0;
        en = 0;

        // Apply reset
        @(negedge clk);
        rst_n = 0;
        en = 0;
        @(negedge clk);
        rst_n = 1; // Deassert reset
        @(negedge clk);

        // Check reset value
        check_count(0, "After reset");

        // Enable counting
        en = 1;

        // Count up to MAX_VALUE and check values
        for (i = 1; i <= MAX_VALUE; i = i + 1) begin
            @(negedge clk);
            check_count(i, $sformatf("Counting up, cycle %0d", i));
            if (i == MAX_VALUE) begin
                if (!wrap) $display("ERROR: wrap signal not asserted at MAX_VALUE (%0d)", MAX_VALUE);
            end else begin
                if (wrap) $display("ERROR: wrap signal asserted prematurely at count=%0d", i);
            end
        end

        // Next clock: should wrap to 0
        @(negedge clk);
        check_count(0, "After wrap-around");
        if (wrap) $display("ERROR: wrap signal should not be asserted after wrap-around");

        // Continue counting a few more cycles
        for (i = 1; i <= 3; i = i + 1) begin
            @(negedge clk);
            check_count(i, $sformatf("Post-wrap counting, cycle %0d", i));
        end

        // Test disabling enable
        en = 0;
        @(negedge clk);
        check_count(3, "Enable low, counter should hold value");

        // Test reset during operation
        en = 1;
        @(negedge clk);
        check_count(4, "Counting resumes after enable");
        rst_n = 0;
        @(negedge clk);
        check_count(0, "Reset asserted during operation");
        rst_n = 1;
        @(negedge clk);
        check_count(0, "After reset deasserted");

        $display("Counter testbench completed at %0t", $time);
        $finish;
    end

endmodule
