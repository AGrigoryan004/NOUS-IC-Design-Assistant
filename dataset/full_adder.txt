// Parameterized Full Adder Module
module full_adder #(
    parameter integer WIDTH = 8 // Number of bits for the adder
)(
    input  wire [WIDTH-1:0] A,      // First operand
    input  wire [WIDTH-1:0] B,      // Second operand
    input  wire              Cin,   // Carry-in
    output wire [WIDTH-1:0] Sum,    // Sum output
    output wire              Cout    // Carry-out
);

    // Internal wire for sum and carry
    wire [WIDTH:0] result;

    // Perform addition with carry-in
    assign result = {1'b0, A} + {1'b0, B} + Cin;

    assign Sum  = result[WIDTH-1:0];
    assign Cout = result[WIDTH];

endmodule
