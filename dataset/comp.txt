module comparator #(
    parameter WIDTH = 8 // Default bit-width is 8
)(
    input  wire [WIDTH-1:0] a,
    input  wire [WIDTH-1:0] b,
    output wire             a_gt_b, // a > b
    output wire             a_eq_b, // a == b
    output wire             a_lt_b  // a < b
);

    // Compare a and b
    assign a_gt_b = (a > b);
    assign a_eq_b = (a == b);
    assign a_lt_b = (a < b);

endmodule
